85 QList<Move> Referee::getQueenJumps(const Location &location, Board &board) {
86  QList<Move> moves;
87 
88  foreach (Location dir, getQueenDirections()) {
89  //qDebug() << dir.toString();
90  Location current_location(location);
91 
92  do {
93  current_location += dir;
94  // posunuli jsme se na další pozici a kontrolujeme zda na ní je protihráč
95  // pokud ano tak skáčeme na libovolnou VOLNOU pozici za tímto kamenem
96  if (current_location.isOnBoard(board) && Figure::getColorByType(board(current_location)) == Figure::negateColor(Figure::getColorByType(board(location)))) {
97  // posunuli jsme se tedy na prvni pozici za protihráčův kámen
98  //qDebug() << board(current_location);
99  Location next_location = current_location + dir;
100  // dokud je tato pozice volná
101  while (next_location.isOnBoard(board) && next_location.isFree(board)) {
102  Move move;
103  move.setFigureType(board(location));
104  move.setFrom(Location(location));
105  move.setTo(Location(next_location));
106  move.addJumpedFigure(current_location, board(current_location));
107  // delame tah a na teto upravene desce hledame dalsi meziskoky z daneho mista
108  board.makeMove(move, false);
109  QList<Move> next_moves = getQueenJumps(next_location, board);
110  // jsou k dispozici dalsi skoky
111  if (next_moves.size() > 0) {
112  foreach (Move next_move, next_moves) {
113  Move final_move(move);
114  final_move.setTo(Location(next_move.getTo()));
115  final_move.addJumpedFigures(next_move.getJumpedFigures());
116  moves.append(final_move);
117  }
118  }
119  else {
120  moves.append(move);
121  }
122  board.makeInverseMove(move, false);
123  next_location += dir;
124  }
125  break;
126  }
127  }
128  while (current_location.isOnBoard(board) && current_location.isFree(board));
129  }
